```cpp
int main() {
    int h, w; cin >> h >> w;
    vector<string> A(h); rep(i, 0, h) cin >> A[i];
    vector<vector<ll>> dp(h, vector<ll>(w, 0));
    dp[0][0] = 1;
    rep(i, 0, h) rep(j, 0, w) {
        if (i+1 < h && A[i+1][j] == '.') {
            dp[i+1][j] += dp[i][j];
            dp[i+1][j] %= 1000000007;
        }
        if (j+1 < w && A[i][j+1] == '.'){
            dp[i][j+1] += dp[i][j];
            dp[i][j+1] %= 1000000007;
        }
    }
    cout << dp[h-1][w-1] << endl;
}
```
