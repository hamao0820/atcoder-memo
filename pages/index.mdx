import { useData } from "nextra/data";
import { getRating } from "api/rating";

export const getStaticProps = async ({ params }) => {
  const data = await getRating();
  return {
    props: { ssg: { ...data } },
    revalidate: 60 * 60 * 24,
  };
};

export const Rating = () => {
  const { rating, colorCode } = useData();
  return <strong style={{ color: colorCode }}>{rating}</strong>;
};

export const HighestRating = () => {
  const { highestRating, highestColorCode } = useData();
  return <strong style={{ color: highestColorCode }}>{highestRating}</strong>;
};
